% 1. Эстетичные списки
% Убираем лишний «воздух», меняем жирные точки на элегантные тире
\setlist{
    leftmargin=*,          % Убираем огромный левый отступ
    topsep=0.3em,          % Уменьшаем отступ перед списком
    itemsep=1em,         % Уменьшаем расстояние между пунктами
    parsep=0.1em           % Уменьшаем расстояние между абзацами внутри пункта
}
\setlist[itemize,1]{label=\textendash} % Первый уровень: короткое тире
\setlist[itemize,2]{label=$\circ$}     % Второй уровень: пустое колечко
\setlist[enumerate,1]{label=\arabic*., font=\sffamily} % Цифры без засечек

% 2. Минималистичные выделения (textbf и textit)
% Сохраняем оригинальные команды
\let\oldtextbf\textbf
\let\oldtextit\textit

% Переопределяем textbf: делаем его на 20% светлее (темно-серым) 
% и переводим в шрифт без засечек (опционально, но выглядит очень свежо)
\renewcommand{\textbf}[1]{{\sffamily\color{black!80}\oldtextbf{#1}}}

% Переопределяем textit: также слегка приглушаем цвет, 
% чтобы курсив не "рябил" в глазах
\renewcommand{\textit}[1]{{\color{black!85}\oldtextit{#1}}}

\usepackage{amsmath, amssymb, amsthm, mathtools} % Базовый стек AMS

\newcommand{\D}{\mathbb{D}}         % Дисперсия (русская нотация)

% Операторы с правильными отступами 
\DeclareMathOperator{\cov}{cov}     % Ковариация
\DeclareMathOperator{\Var}{Var}     % Дисперсия (международная нотация)
\DeclareMathOperator{\Cov}{Cov}     % Ковариация (с заглавной)

% Стандартные распределения
\DeclareMathOperator{\Be}{Be}       % Бернулли
\DeclareMathOperator{\Bi}{Bi}       % Биномиальное
\DeclareMathOperator{\Po}{Po}       % Пуассона
\newcommand{\N}{\mathcal{N}}        % Нормальное распределение

% Символ независимости
\newcommand{\indep}{\perp\!\!\!\perp}

% Виды сходимости 
\newcommand{\toP}{\xrightarrow{\P}}             % По вероятности
\newcommand{\toAS}{\xrightarrow{\text{п.н.}}}   % Почти наверное
\newcommand{\toD}{\xrightarrow{d}}              % По распределению

\usepackage[dvipsnames, table]{xcolor} % Расширенная палитра 
\usepackage[many]{tcolorbox} % Ядро для создания красивых блоков 

% Семантические цвета
\definecolor{MethodBlue}{HTML}{2980B9}
\definecolor{HackGreen}{HTML}{27AE60}
\definecolor{WarnRed}{HTML}{C0392B}

% Базовый стиль блоков (наследует идею notionstyle )
\tcbset{
    profblock/.style={
        enhanced, breakable,
        frame hidden, boxrule=0pt,
        borderline west={3pt}{0pt}{#1}, % Цветная полоса слева 
        colback=#1!5, % Легкий фон (5% от основного цвета) 
        coltitle=#1, fonttitle=\bfseries\sffamily\large,
        attach title to upper={\par\vspace{0.5em}}, % Отступ после заголовка 
        left=12pt, right=12pt, top=12pt, bottom=12pt, % Просторные внутренние отступы 
        arc=0pt % Прямые углы для строгой академичности
    }
}

% ------------------------------------------------------------------------------
% ЭТАП 4: СПЕЦ-ОКРУЖЕНИЯ (БЛОКИ ПРОФЕССОРА)
% ------------------------------------------------------------------------------

% 1. MethodBox (Синий стиль) - Для универсальных алгоритмов
\newtcolorbox{MethodBox}[1][Универсальный алгоритм]{
    profblock=MethodBlue,
    title={ #1}
}

% 2. HackBox (Зеленый стиль) - Для быстрых трюков и эвристик
\newtcolorbox{HackBox}[1][Экзаменационный хак]{
    profblock=HackGreen,
    title={ #1}
}

% 3. WarningBox (Красный/Бордовый стиль) - Для типичных ошибок
\newtcolorbox{WarningBox}[1][Типичная ошибка]{
    profblock=WarnRed,
    title={ #1}
}

\newtcolorbox{Answer}[1][]{
    profblock=AnsRed,
    title={ Ответ\ifstrempty{#1}{}{: #1}}
}
